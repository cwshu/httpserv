102 uint32_t IPv4AddressSet::get_netmask_nbyte(){
103  /* netmask to netmask_nbyte
104  * 0 => 0x000 ... 00 (32 bits uint)
105  * 1 => 0x100 ... 00 (32 bits uint)
106  * 2 => 0x110 ... 00 (32 bits uint)
107  * ...
108  * 32 => 0x111 ... 11 (32 bits uint)
109  *
110  * method:
111  * netmask_hbyte: 1 => 31 => 0x011 ... 11 (2^31 - 1) => 0x100 ... 00 (bitwise not)
112  */
113  if( netmask == 0 ){
114  return 0;
115  }
116 
117  uint32_t reverse_netmask_hbyte = (1u << (32u - netmask)) - 1u;
118  uint32_t netmask_hbyte = ~reverse_netmask_hbyte; // (bitwise not)
119  uint32_t netmask_nbyte = htonl(netmask_hbyte);
120  return netmask_nbyte;
121 }
