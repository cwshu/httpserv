25 std::string fetch_word(std::string& parsed_str, const char* split_chars){
26  /*
27  * fetch the next "word" in parsed_str, and cut this word part in parsed_str.
28  * the "word" means char sequence before split_chars;
29  *
30  * ex.
31  *
32  * split_chars: rn;
33  * parsed_str => return_string, parsed_str_after
34  * "rrrnn" => "" , ""
35  * "rraarnbbbr" => "aa" , "nbbbr"
36  * "rraaabbbrn" => "aaabbb" , ""
37  */
38  std::size_t start = parsed_str.find_first_not_of(split_chars);
39  std::size_t end = parsed_str.find_first_of(split_chars, start);
40  // std::size_t next_start = parsed_str.find_first_not_of(split_chars, end);
41  if( start == std::string::npos ){
42  /* no word, parsed_str only consist of split_chars */
43  parsed_str.clear();
44  return std::string();
45  }
46 
47  std::string word;
48  if( end == std::string::npos ){
49  word = parsed_str.substr(start, std::string::npos);
50  parsed_str.clear();
51  }
52  else{
53  word = parsed_str.substr(start, end - start);
54  parsed_str.erase(0, end + 1);

58 }
